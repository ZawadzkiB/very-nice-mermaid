As-built structure (class)

classDiagram
  class rough["src/rough (source of truth)"] {
    +roughSeed(key) number
    +roughShape(pts, key) fill+outline[]
    +roughPolyline(pts, key) string[]
    +openArrowhead(tip, from, size, key) string
    +ellipsePoints() ; SKETCH consts
  }
  class runtimeRough["vnmRuntime inlined mirror"] {
    +roughSeed / mulberry32 / roughStroke
    +roughShape / roughPolyline / openArrowhead
    +SK_* consts (== SKETCH)
  }
  class sketchFont["sketch-font.ts"] {
    +SKETCH_FONT_FAMILY "Kalam"
    +SKETCH_FONT_BASE64 (woff2)
    +sketchFontFaceCss()
  }
  class sketchSvg["sketch-svg.ts (native helper)"] {
    +sketchRectSvg / sketchEllipseSvg
    +sketchLineSvg / sketchArrowSvg
  }
  rough <.. runtimeRough : MIRROR (byte-parity)
  rough <.. sketchSvg : uses
  sketchFont <.. sketchSvg
  sketchFont <.. runtimeRough : rides payload

Sketch render call (sequence)

sequenceDiagram
  actor User
  participant CLI as vnm render --style sketch
  participant RN as renderer (svg / native / runtime)
  participant RG as src/rough (seeded)
  participant FT as sketch-font (Kalam)
  User->>CLI: render diagram.mmd --style sketch
  CLI->>RN: renderX({ style: "sketch" })
  RN->>RG: roughShape / openArrowhead (seed = id)
  RG-->>RN: wobbly d-strings (deterministic)
  RN->>FT: sketchFontFaceCss() + family
  FT-->>RN: @font-face base64 + family
  RN-->>CLI: SVG (rough + open arrowheads + font)
  Note over RN: PNG → resvg Kalam buffer · HTML → vnmRuntime inlines the same gen (parity)
